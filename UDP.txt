public class timsoconthieu {
    public static void main(String[] args) throws Exception {

        DatagramSocket ketNoi = new DatagramSocket();

        InetAddress diaChiServer =
                InetAddress.getByName("36.50.135.242");

        int congServer = 2207;

        String maGui = ";B21DCCN900;cVxwn5Sv";

        byte[] duLieuGui = maGui.getBytes();

        DatagramPacket goiGui =
                new DatagramPacket(
                        duLieuGui,
                        duLieuGui.length,
                        diaChiServer,
                        congServer
                );

        ketNoi.send(goiGui);

        byte[] duLieuNhan = new byte[1024];

        DatagramPacket goiNhan =
                new DatagramPacket(
                        duLieuNhan,
                        duLieuNhan.length
                );

        ketNoi.receive(goiNhan);

        String chuoiNhan = new String(
                goiNhan.getData(),
                0,
                goiNhan.getLength()
        ).trim();

        System.out.println("Chuoi nhan: " + chuoiNhan);

        String[] cacPhan = chuoiNhan.split(";");

        String maYeuCau = cacPhan[0];

        int n = Integer.parseInt(cacPhan[1].trim());

        String[] danhSachSo = cacPhan[2].split(",");

        boolean[] daXuatHien = new boolean[n + 1];

        for (String so : danhSachSo) {

            int giaTri = Integer.parseInt(so.trim());

            if (giaTri >= 1 && giaTri <= n) {
                daXuatHien[giaTri] = true;
            }
        }

        String soConThieu = "";

        for (int i = 1; i <= n; i++) {

            if (!daXuatHien[i]) {

                if (!soConThieu.isEmpty()) {
                    soConThieu += ",";
                }

                soConThieu += i;
            }
        }

        String ketQua = maYeuCau + ";" + soConThieu;

        System.out.println("Ket qua: " + ketQua);

        byte[] duLieuKetQua = ketQua.getBytes();

        DatagramPacket goiKetQua =
                new DatagramPacket(
                        duLieuKetQua,
                        duLieuKetQua.length,
                        diaChiServer,
                        congServer
                );

        ketNoi.send(goiKetQua);

        ketNoi.close();
    }
}


public class kytuxuathiennhieunhat {
    public static void main(String[] args) throws Exception {

        DatagramSocket socket = new DatagramSocket();

        InetAddress address =
                InetAddress.getByName("36.50.135.242");

        int port = 2208;

        // a. Gửi mã sinh viên và mã câu hỏi
        String code = ";B21DCCN900;E75TpJe9";
        byte[] sendData = code.getBytes();

        DatagramPacket sendPacket = new DatagramPacket(
                sendData,
                sendData.length,
                address,
                port
        );

        socket.send(sendPacket);

        // b. Nhận dữ liệu từ server
        byte[] buffer = new byte[1024];

        DatagramPacket receivePacket =
                new DatagramPacket(buffer, buffer.length);

        socket.receive(receivePacket);

        String received = new String(
                receivePacket.getData(),
                0,
                receivePacket.getLength()
        ).trim();

        System.out.println("Server gửi: " + received);

        // Tách requestId và data
        String[] parts = received.split(";", 2);

        String requestId = parts[0];
        String data = parts[1].trim();

        // c. Đếm số lần xuất hiện của từng ký tự
        int[] count = new int[256];

        for (int i = 0; i < data.length(); i++) {
            char c = data.charAt(i);
            count[c]++;
        }

        // Tìm ký tự xuất hiện nhiều nhất
        char maxCharacter = data.charAt(0);
        int maxCount = count[maxCharacter];

        for (int i = 1; i < data.length(); i++) {
            char c = data.charAt(i);

            if (count[c] > maxCount) {
                maxCount = count[c];
                maxCharacter = c;
            }
        }

        // Tìm các vị trí xuất hiện của ký tự đó
        String positions = "";

        for (int i = 0; i < data.length(); i++) {
            if (data.charAt(i) == maxCharacter) {
                positions += (i + 1) + ",";
            }
        }

        // Tạo kết quả
        String result =
                requestId + ";" + maxCharacter + ":" + positions;

        System.out.println("Gửi lại: " + result);

        // Gửi kết quả lên server
        byte[] resultData = result.getBytes();

        DatagramPacket resultPacket = new DatagramPacket(
                resultData,
                resultData.length,
                address,
                port
        );

        socket.send(resultPacket);

        // d. Đóng socket
        socket.close();
    }
}

public class mahoaceasar {
    public static void main(String[] args) throws Exception {

        // Tạo socket UDP
        DatagramSocket socket = new DatagramSocket();

        // Địa chỉ và cổng server
        InetAddress address =
                InetAddress.getByName("36.50.135.242");

        int port = 2207;

        // a. Gửi mã sinh viên và mã câu hỏi
        String code = ";B21DCCN900;J5SE2YXc";

        byte[] sendData = code.getBytes();

        DatagramPacket sendPacket =
                new DatagramPacket(
                        sendData,
                        sendData.length,
                        address,
                        port
                );

        socket.send(sendPacket);

        // b. Nhận dữ liệu từ server
        byte[] buffer = new byte[1024];

        DatagramPacket receivePacket =
                new DatagramPacket(buffer, buffer.length);

        socket.receive(receivePacket);

        String received = new String(
                receivePacket.getData(),
                0,
                receivePacket.getLength()
        ).trim();

        System.out.println("Server gửi: " + received);

        // Dữ liệu có dạng: requestId;strEncode;s
        String[] parts = received.split(";", 3);

        String requestId = parts[0];
        String strEncode = parts[1];
        int s = Integer.parseInt(parts[2].trim());

        // Đưa độ dịch về khoảng từ 0 đến 25
        s = ((s % 26) + 26) % 26;

        // c. Giải mã chuỗi Caesar
        String strDecode = "";

        for (int i = 0; i < strEncode.length(); i++) {

            char character = strEncode.charAt(i);

            // Nếu là chữ hoa
            if (character >= 'A' && character <= 'Z') {

                char decodedCharacter =
                        (char) ((character - 'A' - s + 26) % 26 + 'A');

                strDecode += decodedCharacter;
            }

            // Nếu là chữ thường
            else if (character >= 'a' && character <= 'z') {

                char decodedCharacter =
                        (char) ((character - 'a' - s + 26) % 26 + 'a');

                strDecode += decodedCharacter;
            }

            // Nếu không phải chữ cái thì giữ nguyên
            else {
                strDecode += character;
            }
        }

        // Tạo kết quả theo định dạng requestId;strDecode
        String result =
                requestId + ";" + strDecode;

        System.out.println("Chuỗi giải mã: " + strDecode);
        System.out.println("Gửi lại: " + result);

        // Gửi kết quả lên server
        byte[] resultData = result.getBytes();

        DatagramPacket resultPacket =
                new DatagramPacket(
                        resultData,
                        resultData.length,
                        address,
                        port
                );

        socket.send(resultPacket);

        // d. Đóng socket
        socket.close();
    }

public class thongtinsinhvien {

    public static String chuanHoaTen(String ten) {

        for (int i = 0; i < cacTu.length; i++) {

            String tu = cacTu[i];

            String tuChuanHoa =
                    tu.substring(0, 1).toUpperCase()
                    + tu.substring(1);

            if (i > 0) {
                ketQua += " ";
            }

            ketQua += tuChuanHoa;
        }

        return ketQua;
    }

    // Hàm tạo email từ tên
    public static String taoEmail(String ten) {

        ten = ten.trim().toLowerCase();

        String[] cacTu = ten.split("\\s+");

        // Từ cuối cùng là tên
        String email = cacTu[cacTu.length - 1];

        // Lấy chữ cái đầu của họ và tên đệm
        for (int i = 0; i < cacTu.length - 1; i++) {
            email += cacTu[i].charAt(0);
        }

        email += "@ptit.edu.vn";

        return email;
    }

        // Tạo socket UDP
        DatagramSocket socket = new DatagramSocket();

        // Địa chỉ và cổng server
        InetAddress diaChi =
                InetAddress.getByName("36.50.135.242");

        int cong = 2209;

        // a. Gửi mã sinh viên và mã câu hỏi
        String maGui = ";B21DCCN900;n1J91Lfu";

        byte[] duLieuGui =

        DatagramPacket goiGui =
                new DatagramPacket(
                        duLieuGui,
                        duLieuGui.length,
                        diaChi,
                        cong
                );

        socket.send(goiGui);

        // b. Nhận dữ liệu từ server
        byte[] boDem = new byte[65535];

        DatagramPacket goiNhan =
                new DatagramPacket(
                        boDem,
                        boDem.length
                );

        socket.receive(goiNhan);

        int doDaiDuLieuNhan = goiNhan.getLength();

        /*
         * 8 byte đầu là requestId
         */
        byte[] requestIdBytes =
                Arrays.copyOfRange(
                        goiNhan.getData(),
                        0,
                        8
                );

        String requestId =
                new String(
                        requestIdBytes,
                        StandardCharsets.UTF_8
                );

        /*
         * Các byte còn lại là đối tượng Student
         */
        byte[] objectBytes =
                Arrays.copyOfRange(
                        goiNhan.getData(),
                        8,
                        doDaiDuLieuNhan
                );

        ByteArrayInputStream byteInput =
                new ByteArrayInputStream(objectBytes);

        ObjectInputStream objectInput =
                new ObjectInputStream(byteInput);

        Student sinhVien =
                (Student) objectInput.readObject();

        objectInput.close();

        System.out.println("Request ID: " + requestId);
        System.out.println("ID: " + sinhVien.getId());
        System.out.println("Tên ban đầu: " + sinhVien.getName());

        // c. Chuẩn hóa tên
        String tenChuanHoa =

        sinhVien.setName(tenChuanHoa);

        sinhVien.setEmail(email);

        /*
         * Chuyển đối tượng Student thành mảng byte
         */
        ByteArrayOutputStream byteOutput =
                new ByteArrayOutputStream();

        ObjectOutputStream objectOutput =
                new ObjectOutputStream(byteOutput);

        objectOutput.writeObject(sinhVien);
        objectOutput.flush();

        byte[] studentBytes =
                byteOutput.toByteArray();

        objectOutput.close();

        /*
         * Tạo dữ liệu gửi:
         * 8 byte đầu là requestId
         * các byte sau là đối tượng Student
         */
        byte[] requestBytes =
                requestId.getBytes(StandardCharsets.UTF_8);

        byte[] duLieuKetQua =
                new byte[8 + studentBytes.length];

        // Sao chép 8 byte requestId vào đầu
        System.arraycopy(
                requestBytes,
                0,
                duLieuKetQua,
                0,
                8
        );

        // Sao chép đối tượng Student vào sau requestId
        System.arraycopy(
                studentBytes,
                0,
                duLieuKetQua,
                8,
                studentBytes.length
        );

        DatagramPacket goiKetQua =
                new DatagramPacket(
                        duLieuKetQua,
                        duLieuKetQua.length,
                        diaChi,
                        cong
                );

        socket.send(goiKetQua);

        // d. Đóng socket
        socket.close();
    }
}

    // Chuẩn hóa một từ:

    public static String vietHoaChuDau(String tu) {

        tu = tu.toLowerCase();

        return tu.substring(0, 1).toUpperCase()
                + tu.substring(1);
    }

    /*
     * Chuẩn hóa tên:
     * nguyen van hai duong
     * ->
     * DUONG, Nguyen Van Hai
     */
    public static String chuanHoaTen(String ten) {

        ten = ten.trim().toLowerCase();

        String[] cacTu = ten.split("\\s+");

        // Từ cuối cùng là tên
        String tenChinh =
                cacTu[cacTu.length - 1].toUpperCase();

        // Chuẩn hóa họ và tên đệm
        String hoVaTenDem = "";

        for (int i = 0; i < cacTu.length - 1; i++) {

            if (i > 0) {
                hoVaTenDem += " ";
            }

            hoVaTenDem += vietHoaChuDau(cacTu[i]);
        }

        return tenChinh + ", " + hoVaTenDem;
    }

    /*
     * Đổi ngày sinh:
     * mm-dd-yyyy
     * ->
     * dd/mm/yyyy
     */
    public static String doiNgaySinh(String ngaySinh) {

        String[] parts = ngaySinh.trim().split("-");

        String thang = parts[0];
        String ngay = parts[1];
        String nam = parts[2];

        return ngay + "/" + thang + "/" + nam;
    }

    /*
     * Tạo tài khoản:
     * nguyen van hai duong
     * ->
     * nvhduong
     */

        taiKhoan += cacTu[cacTu.length - 1];

        return taiKhoan;
    }

     *
     * Ví dụ:
     * T520 thinkpad lenovo
     * ->
     * lenovo thinkpad T520
     */
    public static String suaTenSanPham(String ten) {

        // Xóa khoảng trắng thừa ở đầu và cuối
        ten = ten.trim();

        // Tách tên thành các từ
        String[] cacTu = ten.split("\\s+");

        /*
         * Nếu tên chỉ có một từ thì không cần đổi chỗ
         */
        if (cacTu.length <= 1) {
            return ten;
        }

        // Đổi chỗ từ đầu tiên và từ cuối cùng
        String tam = cacTu[0];

        cacTu[0] = cacTu[cacTu.length - 1];

        cacTu[cacTu.length - 1] = tam;

        // Ghép các từ lại bằng dấu cách
        String ketQua = "";

        for (int i = 0; i < cacTu.length; i++) {

            if (i > 0) {
                ketQua += " ";
            }

            ketQua += cacTu[i];
        }

        return ketQua;
    }

    /*
     * Đảo ngược các chữ số của số lượng.
     *
     * Ví dụ:
     * 9981 -> 1899
     */
    public static int suaSoLuong(int soLuong) {

        int ketQua = 0;

        while (soLuong > 0) {

            // Lấy chữ số cuối cùng
            int chuSoCuoi = soLuong % 10;

            // Thêm chữ số cuối vào kết quả
            ketQua = ketQua * 10 + chuSoCuoi;

            // Bỏ chữ số cuối của số lượng
            soLuong = soLuong / 10;
        }

        return ketQua;
    }

    /*
     * Chuẩn hóa một từ:
     * jaVA -> Java
     */
    public static String vietHoaChuDau(String tu) {

        tu = tu.toLowerCase();

        return tu.substring(0, 1).toUpperCase()
                + tu.substring(1);
    }

    /*
     * Chuẩn hóa tiêu đề:
     *
     * lap TRINH mang may tinh
     * ->
     * Lap Trinh Mang May Tinh
     */
    public static String chuanHoaTieuDe(String tieuDe) {

        tieuDe = tieuDe.trim().toLowerCase();

        String[] cacTu = tieuDe.split("\\s+");

        String ketQua = "";

        for (int i = 0; i < cacTu.length; i++) {

            if (i > 0) {
                ketQua += " ";
            }

            ketQua += vietHoaChuDau(cacTu[i]);
        }

        return ketQua;
    }

    /*
     * Chuẩn hóa tác giả:
     *
     * nguyen van anh
     * ->
     * NGUYEN, Van Anh
     */
    public static String chuanHoaTacGia(String tacGia) {

        tacGia = tacGia.trim().toLowerCase();

        String[] cacTu = tacGia.split("\\s+");

        // Từ đầu tiên là họ, viết hoa toàn bộ
        String ho = cacTu[0].toUpperCase();

        // Các từ còn lại là tên
        String ten = "";

        for (int i = 1; i < cacTu.length; i++) {

            if (i > 1) {
                ten += " ";
            }

            ten += vietHoaChuDau(cacTu[i]);
        }

        /*
         * Nếu chỉ có một từ thì chỉ trả về họ.
         * Bình thường tên tác giả sẽ có từ hai từ trở lên.
         */
        if (cacTu.length == 1) {
            return ho;
        }

        return ho + ", " + ten;
    }

    /*
     * Chuẩn hóa ISBN:
     *
     * 9783161484100
     * ->
     * 978-3-16-148410-0
     */
    public static String chuanHoaISBN(String isbn) {

        /*
         * Xóa toàn bộ ký tự không phải chữ số,
         * ví dụ dấu gạch ngang hoặc khoảng trắng.
         */
        String cacChuSo = isbn.replaceAll("\\D", "");

        /*
         * ISBN theo ví dụ có 13 chữ số:
         *
         * 978 | 3 | 16 | 148410 | 0
         */
        String phan1 = cacChuSo.substring(0, 3);
        String phan2 = cacChuSo.substring(3, 4);
        String phan3 = cacChuSo.substring(4, 6);
        String phan4 = cacChuSo.substring(6, 12);
        String phan5 = cacChuSo.substring(12, 13);

        return phan1 + "-"
                + phan2 + "-"
                + phan3 + "-"
                + phan4 + "-"
                + phan5;
    }

    /*
     * Chuyển ngày xuất bản:
     *
     * yyyy-mm-dd
     * ->
     * mm/yyyy
     *
     * Ví dụ:
     * 2025-11-07
     * ->
     * 11/2025
     */
    public static String doiNgayXuatBan(String ngayXuatBan) {

        String[] cacPhan =
                ngayXuatBan.trim().split("-");

        String nam = cacPhan[0];
        String thang = cacPhan[1];

        return thang + "/" + nam;
    }

     * Chuẩn hóa tên:
     *
     * john DOE
     * ->
     * John Doe
     */
    public static String chuanHoaTen(String ten) {

        ten = ten.trim().toLowerCase();

        String[] cacTu = ten.split("\\s+");

        String ketQua = "";

        for (int i = 0; i < cacTu.length; i++) {

            String tu = cacTu[i];

            String tuChuanHoa =
                    tu.substring(0, 1).toUpperCase()
                    + tu.substring(1);

            if (i > 0) {
                ketQua += " ";
            }

            ketQua += tuChuanHoa;
        }

        return ketQua;
    }

    /*
     * Tính tổng các chữ số của năm sinh.
     *
     * Ví dụ:
     * 2000 -> 2 + 0 + 0 + 0 = 2
     */
    public static int tinhTongChuSoNamSinh(int namSinh) {

        int tong = 0;

        while (namSinh > 0) {

            int chuSoCuoi = namSinh % 10;

            tong += chuSoCuoi;

            namSinh = namSinh / 10;
        }

        return tong;
    }

    /*
     * Tăng lương x%.
     *
     * Ví dụ:
     * lương = 10.000.000
     * x = 2
     *
     * lương mới = 10.000.000 + 10.000.000 * 2 / 100
     */
    public static double tangLuong(
            double luong,
            int phanTram
    ) {
        return luong + luong * phanTram / 100;
    }

    /*
     * Đổi ngày tuyển dụng:
     *
     * yyyy-mm-dd
     * ->
     * dd/mm/yyyy
     */
    public static String doiNgayTuyenDung(String ngay) {

        String[] cacPhan =
                ngay.trim().split("-");

        String nam = cacPhan[0];
        String thang = cacPhan[1];
        String ngayTrongThang = cacPhan[2];

        return ngayTrongThang
                + "/"
                + thang
                + "/"
                + nam;
    }

    public static void main(String[] args) throws Exception {

        // Tạo socket UDP
        DatagramSocket socket = new DatagramSocket();

        // Địa chỉ server
        InetAddress diaChi =
                InetAddress.getByName("36.50.135.242");

        // Cổng của bài đối tượng
        int cong = 2209;

        /*
         * a. Gửi mã sinh viên và mã câu hỏi
         */
        String maGui =
                ";B21DCCN900;ySsumsIE";

        byte[] duLieuGui =
                maGui.getBytes(StandardCharsets.UTF_8);

        DatagramPacket goiGui =
                new DatagramPacket(
                        duLieuGui,
                        duLieuGui.length,
                        diaChi,
                        cong
                );

        socket.send(goiGui);

        /*
         * b. Nhận dữ liệu từ server
         */
        byte[] boDem = new byte[65535];

        DatagramPacket goiNhan =
                new DatagramPacket(
                        boDem,
                        boDem.length
                );

        socket.receive(goiNhan);

        int doDaiDuLieuNhan =
                goiNhan.getLength();

        /*
         * 8 byte đầu là requestId
         */
        byte[] requestIdBytes =
                Arrays.copyOfRange(
                        goiNhan.getData(),
                        0,
                        8
                );

        String requestId =
                new String(
                        requestIdBytes,
                        StandardCharsets.UTF_8
                );

        /*
         * Các byte còn lại là đối tượng Employee
         */
        byte[] employeeBytes =
                Arrays.copyOfRange(
                        goiNhan.getData(),
                        8,
                        doDaiDuLieuNhan
                );

        /*
         * Chuyển byte thành đối tượng Employee
         */
        ByteArrayInputStream byteInput =
                new ByteArrayInputStream(employeeBytes);

        ObjectInputStream objectInput =
                new ObjectInputStream(byteInput);

        Employee nhanVien =
                (Employee) objectInput.readObject();

        objectInput.close();

        /*
         * In thông tin ban đầu để kiểm tra
         */
        System.out.println(
                "Request ID: " + requestId
        );

        System.out.println(
                "ID: " + nhanVien.getId()
        );

        System.out.println(
                "Tên ban đầu: "
                        + nhanVien.getName()
        );

        System.out.println(
                "Lương ban đầu: "
                        + nhanVien.getSalary()
        );

        System.out.println(
                "Ngày tuyển dụng ban đầu: "
                        + nhanVien.getHireDate()
        );

        /*
         * c1. Chuẩn hóa tên
         */
        String tenMoi =
                chuanHoaTen(nhanVien.getName());

        nhanVien.setName(tenMoi);

        /*
         * c2. Tính phần trăm tăng lương
         *
         * Năm sinh của bạn là 2000:
         * 2 + 0 + 0 + 0 = 2
         */
        int namSinh = 2000;

        int phanTramTang =
                tinhTongChuSoNamSinh(namSinh);

        double luongMoi =
                tangLuong(
                        nhanVien.getSalary(),
                        phanTramTang
                );

        nhanVien.setSalary(luongMoi);

        /*
         * c3. Đổi định dạng ngày tuyển dụng
         */
        String ngayTuyenDungMoi =
                doiNgayTuyenDung(
                        nhanVien.getHireDate()
                );

        nhanVien.setHireDate(ngayTuyenDungMoi);

        /*
         * In thông tin sau khi xử lý
         */
        System.out.println(
                "Tên sau chuẩn hóa: "
                        + nhanVien.getName()
        );

        System.out.println(
                "Phần trăm tăng lương: "
                        + phanTramTang + "%"
        );

        System.out.println(
                "Lương sau khi tăng: "
                        + nhanVien.getSalary()
        );

        System.out.println(
                "Ngày tuyển dụng mới: "
                        + nhanVien.getHireDate()
        );

        /*
         * Chuyển đối tượng Employee đã sửa thành byte
         */
        ByteArrayOutputStream byteOutput =
                new ByteArrayOutputStream();

        ObjectOutputStream objectOutput =
                new ObjectOutputStream(byteOutput);

        objectOutput.writeObject(nhanVien);
        objectOutput.flush();

        byte[] doiTuongKetQua =
                byteOutput.toByteArray();

        objectOutput.close();

        /*
         * Tạo dữ liệu gửi lại:
         *
         * 8 byte đầu là requestId
         * Các byte sau là đối tượng Employee
         */
        byte[] requestBytes =
                requestId.getBytes(StandardCharsets.UTF_8);

        byte[] duLieuKetQua =
                new byte[8 + doiTuongKetQua.length];

        // Sao chép 8 byte requestId
        System.arraycopy(
                requestBytes,
                0,
                duLieuKetQua,
                0,
                8
        );

        // Sao chép đối tượng Employee vào phía sau
        System.arraycopy(
                doiTuongKetQua,
                0,
                duLieuKetQua,
                8,
                doiTuongKetQua.length
        );

        /*
         * Gửi dữ liệu đã xử lý về server
         */
        DatagramPacket goiKetQua =
                new DatagramPacket(
                        duLieuKetQua,
                        duLieuKetQua.length,
                        diaChi,
                        cong
                );

        socket.send(goiKetQua);

        /*
         * d. Đóng socket
         */
        socket.close();
    }
}

import java.io.Serializable;

public class Employee implements Serializable {

    private static final long serialVersionUID = 20261107L;

    private String id;
    private String name;
    private double salary;
    private String hireDate;

    public Employee(
            String id,
            String name,
            double salary,
            String hireDate
    ) {
        this.id = id;
        this.name = name;
        this.salary = salary;
        this.hireDate = hireDate;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public double getSalary() {
        return salary;
    }

    public void setSalary(double salary) {
        this.salary = salary;
    }

    public String getHireDate() {
        return hireDate;
    }

    public void setHireDate(String hireDate) {
        this.hireDate = hireDate;
    }
}
