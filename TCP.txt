public class tongcacsonguyento {

    // Hàm kiểm tra một số có phải số nguyên tố hay không
    public static boolean isPrime(int n) {

        // Các số nhỏ hơn 2 không phải số nguyên tố
        if (n < 2) {
            return false;
        }

        // Chỉ cần kiểm tra từ 2 đến căn bậc hai của n
        for (int i = 2; i * i <= n; i++) {

            // Nếu n chia hết cho i thì n không phải số nguyên tố
            if (n % i == 0) {
                return false;
            }
        }

        // Không tìm thấy ước nào khác 1 và chính nó
        return true;
    }

    public static void main(String[] args) throws Exception {
        Socket socket = new Socket("36.50.135.242", 2206);
        InputStream in = socket.getInputStream();
        OutputStream out = socket.getOutputStream();
        String code = "B21DCCN900;E8XZy0VE";
        out.write(code.getBytes());
        out.flush();
        byte[] buffer = new byte[1024];

        int n = in.read(buffer);
        String s = new String(buffer, 0, n).trim();
        
        System.out.println("Chuoi nhan duoc: " + s);
        String[] parts = s.split(",");
        int sum = 0;

        // Duyệt từng số trong chuỗi
        for (int i = 0; i < parts.length; i++) {

            // Chuyển chuỗi thành số nguyên
            int number = Integer.parseInt(parts[i].trim());

            // Nếu là số nguyên tố thì cộng vào tổng
            if (isPrime(number)) {
                sum += number;
            }
        }

        System.out.println("Tong cac so nguyen to: " + sum);
        String result = String.valueOf(sum);
        
        out.write(result.getBytes());
        out.flush();
        socket.close();
    }
}

        BufferedWriter out = new BufferedWriter(
                new OutputStreamWriter(socket.getOutputStream())
        );

        // Mã sinh viên và mã câu hỏi
        String code = "B21DCCN900;Ov12N2xM";

        // Gửi mã sinh viên và mã câu hỏi
        out.write(code);
        out.newLine();
        out.flush();

        // Nhận chuỗi từ server
        String s = in.readLine();

        System.out.println("Chuoi nhan duoc: " + s);

        // Tách chuỗi thành các từ
        String[] words = s.trim().split("\\s+");

        // Sắp xếp các từ theo độ dài tăng dần

            for (int j = 0; j < words.length - 1 - i; j++) {

                if (words[j].length() > words[j + 1].length()) {

                    String temp = words[j];
                    words[j] = words[j + 1];
                    words[j + 1] = temp;
                }
            }
        }
        // Ghép các từ thành chuỗi kết quả
        String result = "";

        for (int i = 0; i < words.length; i++) {

            result += words[i];

            // Không thêm dấu phẩy sau từ cuối cùng
            if (i < words.length - 1) {
                result += ", ";
            }
        }
        System.out.println("Ket qua: " + result);

        // Gửi kết quả lên server
        out.write(result);
        out.newLine();
        out.flush();

        // Đóng kết nối
        socket.close();

public class tonggapdoitrungbinh {
    public static void main(String[] args) throws Exception {

        // Kết nối tới server
        Socket socket = new Socket("36.50.135.242", 2206);

        // Luồng nhận dữ liệu từ server
        InputStream in = socket.getInputStream();

        // Luồng gửi dữ liệu lên server
        OutputStream out = socket.getOutputStream();

        // Mã sinh viên và mã câu hỏi
        String code = "B21DCCN900;E8XZy0VE";

        // Gửi mã sinh viên và mã câu hỏi
        out.write(code.getBytes());
        out.flush();

        // Tạo mảng byte để nhận dữ liệu
        byte[] buffer = new byte[1024];

        // Nhận dữ liệu từ server
        int n = in.read(buffer);

        // Chuyển dữ liệu byte thành chuỗi
        String s = new String(buffer, 0, n).trim();

        System.out.println("Chuoi nhan duoc: " + s);

        // Tách chuỗi theo dấu phẩy
        String[] parts = s.split(",");

        // Chuyển mảng String thành mảng int
        int[] numbers = new int[parts.length];

        int sum = 0;

        for (int i = 0; i < parts.length; i++) {
            numbers[i] = Integer.parseInt(parts[i].trim());
            sum += numbers[i];
        }

        // Gấp đôi trung bình:
        // 2 * trung bình = 2 * sum / số phần tử
        double target = 2.0 * sum / numbers.length;

        System.out.println("Gap doi trung binh: " + target);

        // Khởi tạo cặp đầu tiên
        int num1 = numbers[0];
        int num2 = numbers[1];

        // Độ chênh lệch nhỏ nhất ban đầu
        double minDifference = Double.MAX_VALUE;

        // Duyệt tất cả các cặp số khác nhau
        for (int i = 0; i < numbers.length - 1; i++) {

            for (int j = i + 1; j < numbers.length; j++) {

                double difference =
                        Math.abs(numbers[i] + numbers[j] - target);

                if (difference < minDifference) {
                    minDifference = difference;
                    num1 = numbers[i];
                    num2 = numbers[j];
                }
            }
        }

        // Tạo chuỗi kết quả
        String result = num1 + "," + num2;

        System.out.println("Ket qua: " + result);

        // Gửi kết quả lên server
        out.write(result.getBytes());
        out.flush();

        // Đóng kết nối
        socket.close();
    }
}

        out.writeUTF(code);
        out.flush();

        String s =
                in.readUTF();

        int shift =
                in.readInt();

        System.out.println("Chuoi ma hoa: " + s);
        System.out.println("Do dich: " + shift);

        String result =
                "";

        for (int i = 0; i < s.length(); i++) {

            char c =
                    s.charAt(i);

            if (c >= 'A' && c <= 'Z') {

                c =
                        (char) ((c - 'A' - shift + 26) % 26 + 'A');

            } else if (c >= 'a' && c <= 'z') {

                c =
                        (char) ((c - 'a' - shift + 26) % 26 + 'a');
            }

            result += c;
        }

        System.out.println("Giai ma: " + result);

        out.writeUTF(result);
        out.flush();

        socket.close();
    }

public class daonguocdoandaiK {
    public static void main(String[] args) throws Exception {

        // Kết nối tới server
        Socket socket = new Socket("36.50.135.242", 2207);

        // Luồng nhận dữ liệu từ server
        DataInputStream in =
                new DataInputStream(socket.getInputStream());

        // Luồng gửi dữ liệu lên server
        DataOutputStream out =
                new DataOutputStream(socket.getOutputStream());

        // Mã sinh viên và mã câu hỏi
        String code = "B21DCCN900;dCNDHojG";

        // Gửi mã sinh viên và mã câu hỏi
        out.writeUTF(code);
        out.flush();

        // Nhận số nguyên k trước
        int k = in.readInt();

        // Nhận chuỗi chứa mảng số nguyên sau
        String s = in.readUTF();

        System.out.println("k = " + k);
        System.out.println("Mang nhan duoc: " + s);

        // Tách chuỗi thành các phần tử
        String[] parts = s.split(",");

        // Chuyển mảng String thành mảng int
        int[] numbers = new int[parts.length];

        for (int i = 0; i < parts.length; i++) {
            numbers[i] = Integer.parseInt(parts[i].trim());
        }

        // Đảo ngược từng đoạn dài k
        for (int start = 0; start < numbers.length; start += k) {

            int left = start;

            int right = Math.min(
                    start + k - 1,
                    numbers.length - 1
            );

            while (left < right) {
                int temp = numbers[left];
                numbers[left] = numbers[right];
                numbers[right] = temp;

                left++;
                right--;
            }
        }

        // Tạo chuỗi kết quả
        String result = "";

        for (int i = 0; i < numbers.length; i++) {
            result += numbers[i];

            if (i < numbers.length - 1) {
                result += ",";
            }
        }

        System.out.println("Ket qua: " + result);

        // Gửi chuỗi kết quả lên server
        out.writeUTF(result);
        out.flush();

        // Đóng kết nối
        socket.close();
    }
}

        DataOutputStream out =
                new DataOutputStream(socket.getOutputStream());

        // Mã sinh viên và mã câu hỏi
        String code = "B21DCCN900;dCNDHojG";

        // Gửi mã sinh viên và mã câu hỏi
        out.writeUTF(code);
        out.flush();

        // Nhận số nguyên k trước
        int k = in.readInt();

        // Nhận chuỗi chứa mảng số nguyên sau
        String s = in.readUTF();

        System.out.println("k = " + k);
        System.out.println("Mang nhan duoc: " + s);

        // Tách chuỗi thành các phần tử
        String[] parts = s.split(",");

        // Chuyển mảng String thành mảng int
        int[] numbers = new int[parts.length];

        for (int i = 0; i < parts.length; i++) {
            numbers[i] = Integer.parseInt(parts[i].trim());
        }

        // Đảo ngược từng đoạn dài k
        for (int start = 0; start < numbers.length; start += k) {

            int left = start;

            int right = Math.min(
                    start + k - 1,
                    numbers.length - 1
            );

            while (left < right) {
                int temp = numbers[left];
                numbers[left] = numbers[right];
                numbers[right] = temp;

                left++;
                right--;
            }
        }

        // Tạo chuỗi kết quả
        String result = "";

        for (int i = 0; i < numbers.length; i++) {
            result += numbers[i];

            if (i < numbers.length - 1) {
                result += ",";
            }
        }

        System.out.println("Ket qua: " + result);

        // Gửi chuỗi kết quả lên server
        out.writeUTF(result);
        out.flush();

        // Đóng kết nối
        socket.close();
    }

public class NenRLE {
    public static String xauDao(String s){
        String res = "";
        for(int i = s.length() - 1;i>=0;i--) res+=String.format("%c", s.charAt(i));
        return res;
    }
    public static String RLE(String s){
        int cnt = 1;
        String ans = "";
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i - 1))cnt++;
            else {
                ans+=String.format("%c", s.charAt(i - 1));
                if(cnt >=2) ans+= String.format("%d", cnt);
                cnt = 1;
            }
        }
        ans+=String.format("%c", s.charAt(s.length() - 1));
        if(cnt >=2) ans+= String.format("%d", cnt);
        return ans;
    }

public class sualaptop {

    // Đổi lại từ đầu tiên và từ cuối cùng
    public static String repairName(String name) {

        String[] words = name.trim().split("\\s+");

        if (words.length <= 1) {
            return name.trim();
        }

        String temp = words[0];
        words[0] = words[words.length - 1];
        words[words.length - 1] = temp;

        String result = "";

        for (int i = 0; i < words.length; i++) {
            result += words[i];

            if (i < words.length - 1) {
                result += " ";
            }
        }

        return result;
    }

    // Đảo ngược các chữ số của số lượng
    public static int repairQuantity(int quantity) {

        int reversed = 0;

        while (quantity > 0) {
            int digit = quantity % 10;
            reversed = reversed * 10 + digit;
            quantity /= 10;
        }

        return reversed;
    }

    public static void main(String[] args) throws Exception {

        Socket socket = new Socket("36.50.135.242", 2209);

        ObjectOutputStream out =
                new ObjectOutputStream(socket.getOutputStream());

        ObjectInputStream in =
                new ObjectInputStream(socket.getInputStream());

        String studentCode = "B21DCCN900;xUKwnjAM";

        // Gửi đối tượng String
        out.writeObject(studentCode);
        out.flush();

        // Nhận đối tượng Laptop
        Laptop laptop = (Laptop) in.readObject();

        System.out.println("Truoc khi sua: " + laptop);

        // Sửa tên
        String repairedName = repairName(laptop.getName());
        laptop.setName(repairedName);

        // Sửa số lượng
        int repairedQuantity = repairQuantity(laptop.getQuantity());
        laptop.setQuantity(repairedQuantity);

        System.out.println("Sau khi sua: " + laptop);

        // Gửi lại đối tượng Laptop đã sửa

    }
